16   * The class is the main GUI class.
17   */
18  public class GUI implements Runnable {
19  
20      private JFrame frame;
21      private Game game;
22  
23      public GUI() {
24          this.game = new Game(8, 10);
25      }
26  
27      @Override
28      public void run() {
29          frame = new JFrame("Minesweeper");
30  
31          frame.setPreferredSize(new Dimension(600, 620));
32          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
33  
34          createComponents(frame.getContentPane());
35  
36          frame.pack();
37          frame.setVisible(true);
38      }
39  
40      private void createComponents(Container container) {
41          int size = this.game.getSizeOfGrid();
42          container.setLayout(new BorderLayout());
43  
44          //a new window instead of JLabel?
45          //"congrats - new game?" or "congrats - size?
46          JLabel statusDisplay = new JLabel("Playing", SwingConstants.CENTER);
47          JPanel grid = new JPanel(new GridLayout(size, size, 3, 3));
48          
49          for (int y = 0; y < size; y++) {
50              for (int x = 0; x < size; x++) {
51                  JButton button = new JButton();
52                  button.setActionCommand(String.valueOf(x) + ":" + String.valueOf(y));
53                  button.addActionListener(new TurnListener(game, button, statusDisplay));
54                  grid.add(button);
55              }
56          }
57          
58          container.add(statusDisplay, BorderLayout.NORTH);
59          container.add(grid, BorderLayout.CENTER);
60  
61      }
62  
63  }
