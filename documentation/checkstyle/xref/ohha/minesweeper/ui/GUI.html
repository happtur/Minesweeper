3   import ohha.minesweeper.ui.frame.MainFrame;
4   import ohha.minesweeper.ui.frame.DisplayPanel;
5   import ohha.minesweeper.ui.grid.TurnListener;
6   import ohha.minesweeper.ui.grid.FlagListener;
7   import ohha.minesweeper.ui.grid.ButtonGrid;
8   import ohha.minesweeper.ui.dialog.GameChoicesDialog;
9   import ohha.minesweeper.logic.Game;
10  import javax.swing.JButton;
11  import javax.swing.JFrame;
12  import javax.swing.JOptionPane;
13  
14  /**
15   * The class is the main GUI class.
16   */
17  public class GUI implements Runnable {
18  
19      private MainFrame frame;
20      private Game game;
21      private DisplayPanel display;
22      private ButtonGrid grid;
23      private GameChoicesDialog dialog;
24      private boolean first;
25  
26      @Override
27      public void run() {
28          game = new Game(8, 10);
29          first = true;
30  
31          display = new DisplayPanel(game.amountOfBombs());
32          this.createGrid();
33          grid.setToolTipTextForButtons("Left-click turns, right-click flags");
34  
35          frame = new MainFrame(display, grid, this);
36          frame.setVisible(true);
37      }
38  
39      /**
40       * The method starts and displays a new game.
41       *
42       */
43      public void newGame() {
44  
45          first = true;
46          this.createGame();
47  
48          this.createGrid();
49          frame.replaceButtonGrid(grid);
50  
51          display.setBombs(game.amountOfBombs());
52  
53      }
54  
55      private void createGrid() {
56  
57          grid = new ButtonGrid(game.getSizeOfGrid());
58  
59          TurnListener tListener = new TurnListener(this);
60          grid.addActionListenerToButtons(tListener);
61  
62          FlagListener fListener = new FlagListener(this);
63          grid.addMouseListenerToButtons(fListener);
64      }
65  
66      private void createGame() {
67  
68          if (this.dialog == null) {
69              dialog = new GameChoicesDialog(frame);
70          }
71          dialog.setVisible(true);
72          int size = dialog.getGridSize();
73          int bombs = dialog.getBombAmount();
74          dialog.setVisible(false);
75  
76          this.game = new Game(size, bombs);
77      }
78  
79      /**
80       * The method performs the actions that are to be performed when the player
81       * flags/unflags a tile. I.e. the tile, display and button are updated
82       *
83       * @param button the button to be flagged
84       */
85      public void actionFlagTile(JButton button) {
86          String command = button.getActionCommand();
87  
88          int[] coordinates = game.toCoordinates(command);
89  
90          if (game.flag(coordinates[0], coordinates[1])) {
91              //setIcon
92              button.setText("x");
93              display.oneLessBomb();
94          } else {
95              //removeIcon
96              button.setText("");
97              display.oneMoreBomb();
98          }
99      }
100 
101     /**
102      * The method performs the actions that are to be performed when the player
103      * attempts to turn a tile.
104      *
105      * @param command the coordinates of the tile in question as String ("x:y")
106      */
107     public void actionTurnTile(String command) {
108         int[] coordinates = game.toCoordinates(command);
109         if (first) {
110             display.setStatus("Playing");
111             first = false;
112         }
113         if (!game.turn(coordinates[0], coordinates[1])) {
114             gameIsOver(command, coordinates);
115 
116         } else if (game.isTurned(coordinates[0], coordinates[1])) {
117             int value = game.getValue(coordinates[0], coordinates[1]);
118             grid.setTextOfButton(command, String.valueOf(value));
119             grid.setButtonAsEnabled(command, false);
120         }
121     }
122 
123     private void gameIsOver(String command, int[] coordinates) {
124         if (game.isWon()) {
125             display.setStatus("Congratulations, you won!");
126             grid.setTextOfButton(command, String.valueOf(game.getValue(coordinates[0], coordinates[1])));
127             grid.setAllButtonsAsEnabled(false);
128 
129         } else {
130             //show the hidden bombs?
131             display.setStatus("You lost");
132             grid.setAllButtonsAsEnabled(false);
133         }
134 
135         if (this.askIfNewGame()) {
136             this.newGame();
137         }
138     }
139 
140     private boolean askIfNewGame() {
141 
142         int answer = JOptionPane.showConfirmDialog(frame,
143                 "Would you like to start a new game?",
144                 "New Game",
145                 JOptionPane.YES_NO_OPTION,
146                 JOptionPane.QUESTION_MESSAGE,
147                 null);
148 
149         return answer == JOptionPane.YES_OPTION;
150     }
151 
152     public JFrame getFrame() {
153         return this.frame;
154     }
155 }
