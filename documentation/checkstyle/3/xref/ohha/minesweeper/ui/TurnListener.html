1   package ohha.minesweeper.ui;

3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   import javax.swing.JButton;
6   import javax.swing.JLabel;
7   import javax.swing.JPanel;
8   import ohha.minesweeper.logic.Game;
9   
10  /**
11   * The class listens and reacts to left-clicks on the tile-buttons.
12   */
13  public class TurnListener implements ActionListener {
14  
15      private Game game;
16      //get it from source? but then Object - JButton
17      private JButton button;
18      private DisplayPanel display;
19      private GUI gui;
20      private boolean first;
21  
22      /**
23       * The constructor.
24       * 
25       * @param game the current game
26       * @param button the button the listener listens to
27       * @param display the game's status display
28       * @param gui the main ui
29       */
30      public TurnListener(Game game, JButton button, DisplayPanel display, GUI gui) {
31          this.game = game;
32          this.button = button;
33          this.display = display;
34          this.gui = gui;
35          this.first = true;
36      }
37  
38      //stuff here or 'repaint'(?) grid or sthg? (especially if 0 -> turn surrounding)
39      @Override
40      public void actionPerformed(ActionEvent e) {
41          String command = e.getActionCommand();
42          int[] coordinates = game.toCoordinates(command);
43          if (first) {
44              display.setStatus("Playing");
45              first = false;
46          }
47          if (!game.turn(coordinates[0], coordinates[1])) {
48              if (game.isWon()) {
49                  display.setStatus("Congratulations, you won!");
50              } else {
51                  display.setStatus("You lost");
52              }
53              gui.newGame(display);
54          } else if (game.isTurned(coordinates[0], coordinates[1])) {
55              button.setText(String.valueOf(game.getValue(coordinates[0], coordinates[1])));
56              button.setEnabled(false);
57          }
58      }
59  
60  }
